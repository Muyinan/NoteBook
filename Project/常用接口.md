#### 3. 迷之代码:

```lua
_G.lua_pairs = _G.lua_pairs or pairs
function _G.pairs (t)
    if type(t) == "userdata" then
        local mt = getmetatable(t)
        return mt.inext, t, nil
    else
        return _G.lua_pairs(t)
    end
end
```
